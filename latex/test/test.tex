	\section{多行公式}
	
	带编号的公式如:
	\begin{gather}
		3+4=5\\
		a+b=c
	\end{gather}
	不带符号的公式如：
	\begin{gather*}
	3+4=5\\
	a+b=c\\
	\end{gather*}
	多行不等式有：
	\begin{equation}
		\begin{split}
		\cos 2x &= \cos^2 x -\sin^2 x \\
		&=2\cos^2 x - 1
		\end{split}
	\end{equation}
	花括号大公式有：
	\begin{equation}
		D(x)=\begin{cases}
		1,&\text{如果} x \in \mathbb{Q};\\
		0,&\text{如果} x \in \mathbb{R}\setminus\mathbb{Q};
		\end{cases}
	\end{equation}
